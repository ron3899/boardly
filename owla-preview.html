2. Mock Authentication (mock-data.ts)

// Hardcoded credentials const MOCK_EMAIL = 'demo@boardly.com' const MOCK_PASSWORD = 'demo1234' const MOCK_TOKEN = 'mock-auth-token-boardly' // Login with validation login: async (input) => { if (input.email === MOCK_EMAIL && input.password === MOCK_PASSWORD) { isAuthenticated = true localStorage.setItem('mock-auth-token', MOCK_TOKEN) return { user: currentUser, token: MOCK_TOKEN } } throw new Error('Invalid email or password') } // /auth/me checks localStorage me: async () => { const token = localStorage.getItem('mock-auth-token') if (token === MOCK_TOKEN) { return { user: currentUser } } throw new Error('Not authenticated') }

3. Middleware (middleware.ts)

// Before: Checked environment variable const useMockApi = process.env.NEXT_PUBLIC_USE_MOCK_API === 'true' // After: Always enabled in mock mode const useMockApi = true if (useMockApi) { // Skip server-side auth - handled client-side return NextResponse.next() }
